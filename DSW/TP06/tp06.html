Função Declarada:

function mensagem() {
alert("Essa mensagem é exibida depois de 3 segundos");
}

Função Anônima:

//Anônima
const msg_exp = function () {
return alert("Essa mensagem é exibida depois de 6 segundos");
}
setTimeout(msg_exp, 6000);


Função Expressão:

//Expressão
const obj = {
alert() {
alert("Essa mensagem é exibida depois de 9 segundos");
}
};
setTimeout(obj.alert, 9000);


Função Arrow:

//ArroW
setTimeout(() => {
alert("Essa mensagem é exibida depois de 12 segundos");
}, 12000);
